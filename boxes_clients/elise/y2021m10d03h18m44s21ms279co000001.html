nbBoules = 5

class Boule {

constructor(txt,color) {
this.txt = txt 

 this.sL = `<span style='font-size:25px; background-color:${color}; color:white; padding:5px; margin:3px; border-radius:25px; font-weight:900; border: solid 2px; border-color:${color};'>`

this. sR = `</span>` }

toString() {
return this.sL +this.txt +this.sR }

}

let Boules = []

for (i=1;i<10;i++) Boules.push(new Boule(‘0’+i,’blue’))

for (i=10;i<50;i++) Boules.push(new Boule(i,’blue’))


let BoulesComplementaires = []

for (i=1;i<10;i++) BoulesComplementaires.push( new Boule(‘0’+i,’red’))

BoulesComplementaires.push( new Boule(‘10’,’red’))

let tirage = []

tirage.sort()
